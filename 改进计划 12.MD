## 🎯 设计原则

### 原则 1: 单一职责

每个模块只负责一个功能：
- `FordAPIService` - 仅负责Ford API调用
- `GMAPIService` - 仅负责GM API调用
- `VehicleAPIProtocol` - 仅定义统一接口
- `TokenValidationService` - 仅负责Token验证
- `SoundFeedbackService` - 仅负责声音播放
- `ChartDataService` - 仅负责图表数据准备

### 原则 2: 高内聚低耦合

模块间通过清晰的接口通信：
```
HomeViewModel → VehicleAPIProtocol (依赖注入)
VehicleAPIProtocol → FordAPIService (实现)
VehicleAPIProtocol → GMAPIService (实现)
VehicleAPIProtocol → TeslaAPIService (实现)
```

### 原则 3: 代码复用

优先复用现有模块：
- ✅ `TeslaAPIService` 作为参考实现
- ✅ `VehicleModel` 复用（已有apiToken字段）
- ✅ `VehicleService` 复用（已有API检测方法）
- ✅ `StatisticsService` 复用（已有数据结构）
- ✅ `NotificationService` 复用（已有通知机制）

### 原则 4: 渐进式增强

- Ford API → 独立模块，不影响Tesla功能
- GM API → 独立模块，不影响Tesla功能
- 统一接口 → 抽象层，向后兼容
- Token检测 → 增强现有认证流程

### 原则 5: 风格一致

遵循现有代码风格：
- SwiftUI + Combine 架构
- MVVM 模式
- 英文命名（符合北美市场）
- 深色主题 + 橙色强调色

### 原则 6: 继承优先

- API类型设置 → 继承现有AppSettings
- Token存储 → 复用现有Keychain机制
- 错误处理 → 继承现有APIError枚举

**需求描述**: 实现Ford车辆远程启动功能，支持FordPass API

**生成规则**:

1. **参考Tesla API架构**
   - 使用相同的错误处理模式
   - 使用相同的认证流程
   - 使用相同的气候控制接口

2. **实现Ford API调用**
   - 认证：FordPass OAuth
   - 启动：`/vehicles/{id}/engine/start`
   - 停止：`/vehicles/{id}/engine/stop`
   - 状态：`/vehicles/{id}/status`

3. **遵循现有代码风格**
   - 使用async/await
   - 使用@MainActor
   - 使用Combine发布状态

**代码示例**:

```swift
// FordAPIService.swift

import Foundation

/// Ford API Service for remote vehicle control
/// Implements VehicleAPIProtocol for Ford vehicles
@MainActor
final class FordAPIService: ObservableObject {
    
    // MARK: - Singleton
    
    static let shared = FordAPIService()
    private init() {}
    
    // MARK: - Published Properties
    
    @Published var isAuthenticated: Bool = false
    @Published var currentVehicleId: String?
    
    // MARK: - API Configuration
    
    private let baseURL = "https://usapi.cv.ford.com/api"
    private let clientID = "YOUR_FORD_CLIENT_ID"
    private let clientSecret = "YOUR_FORD_CLIENT_SECRET"
    
    // MARK: - Authentication
    
    /// Authenticate with FordPass
    /// - Parameters:
    ///   - email: User's FordPass email
    ///   - password: User's FordPass password
    /// - Throws: APIError on authentication failure
    func authenticate(email: String, password: String) async throws {
        let authURL = URL(string: "\(baseURL)/oauth/v2/token")!
        
        var request = URLRequest(url: authURL)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        
        let body: [String: Any] = [
            "grant_type": "password",
            "client_id": clientID,
            "client_secret": clientSecret,
            "username": email,
            "password": password
        ]
        
        request.httpBody = try? JSONSerialization.data(withJSONObject: body)
        
        let (data, response) = try await URLSession.shared.data(for: request)
        
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            throw APIError.authenticationFailed
        }
        
        guard let json = try JSONSerialization.jsonObject(with: data) as? [String: Any],
              let accessToken = json["access_token"] as? String else {
            throw APIError.invalidResponse
        }
        
        // Store token in Keychain
        try KeychainHelper.saveToken(accessToken, for: "ford")
        isAuthenticated = true
    }
    
    // MARK: - Climate Control
    
    /// Start climate control (remote start)
    /// - Throws: APIError on failure
    func startClimate() async throws {
        guard let token = try? KeychainHelper.getToken(for: "ford"),
              let vehicleId = currentVehicleId else {
            throw APIError.notAuthenticated
        }
        
        let url = URL(string: "\(baseURL)/vehicles/\(vehicleId)/engine/start")!
        
        var request = URLRequest(url: url)
        request.httpMethod = "POST"
        request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        
        let (_, response) = try await URLSession.shared.data(for: request)
        
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            throw APIError.commandFailed
        }
    }
    
    /// Stop climate control
    /// - Throws: APIError on failure
    func stopClimate() async throws {
        guard let token = try? KeychainHelper.getToken(for: "ford"),
              let vehicleId = currentVehicleId else {
            throw APIError.notAuthenticated
        }
        
        let url = URL(string: "\(baseURL)/vehicles/\(vehicleId)/engine/stop")!
        
        var request = URLRequest(url: url)
        request.httpMethod = "POST"
        request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        
        let (_, response) = try await URLSession.shared.data(for: request)
        
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            throw APIError.commandFailed
        }
    }
    
    /// Get climate state
    /// - Returns: ClimateState with current status
    /// - Throws: APIError on failure
    func getClimateState() async throws -> ClimateState {
        guard let token = try? KeychainHelper.getToken(for: "ford"),
              let vehicleId = currentVehicleId else {
            throw APIError.notAuthenticated
        }
        
        let url = URL(string: "\(baseURL)/vehicles/\(vehicleId)/status")!
        
        var request = URLRequest(url: url)
        request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        
        let (data, response) = try await URLSession.shared.data(for: request)
        
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            throw APIError.networkError
        }
        
        let decoder = JSONDecoder()
        let climateResponse = try decoder.decode(FordClimateResponse.self, from: data)
        
        return ClimateState(
            isClimateOn: climateResponse.engineStatus == "running",
            insideTemp: climateResponse.cabinTemperature,
            outsideTemp: climateResponse.ambientTemperature
        )
    }
}

// MARK: - Ford API Models

private struct FordClimateResponse: Codable {
    let engineStatus: String
    let cabinTemperature: Double?
    let ambientTemperature: Double?
}

// MARK: - Error Types

extension FordAPIService {
    enum APIError: LocalizedError {
        case notAuthenticated
        case authenticationFailed
        case commandFailed
        case networkError
        case invalidResponse
        
        var errorDescription: String? {
            switch self {
            case .notAuthenticated:
                return "Not authenticated. Please login."
            case .authenticationFailed:
                return "Authentication failed. Please check your credentials."
            case .commandFailed:
                return "Failed to execute command. Please try again."
            case .networkError:
                return "Network error. Please check your connection."
            case .invalidResponse:
                return "Invalid response from server."
            }
        }
    }
}
```

```swift
// VehicleAPIProtocol.swift

import Foundation

/// Unified Vehicle API Protocol
/// Abstracts different vehicle brand APIs into a single interface
protocol VehicleAPIProtocol {
    
    // MARK: - Authentication
    
    /// Authenticate with vehicle manufacturer's API
    /// - Parameters:
    ///   - email: User's email
    ///   - password: User's password
    /// - Throws: VehicleAPIError on authentication failure
    func authenticate(email: String, password: String) async throws
    
    // MARK: - Climate Control
    
    /// Start climate control (remote start)
    /// - Throws: VehicleAPIError on failure
    func startClimate() async throws
    
    /// Stop climate control
    /// - Throws: VehicleAPIError on failure
    func stopClimate() async throws
    
    // MARK: - Status
    
    /// Get current climate state
    /// - Returns: ClimateState with current status
    /// - Throws: VehicleAPIError on failure
    func getClimateState() async throws -> ClimateState
}

// MARK: - Climate State Model

/// Unified climate state model
struct ClimateState {
    let isClimateOn: Bool
    let insideTemp: Double?
    let outsideTemp: Double?
}

// MARK: - Error Types

/// Unified vehicle API error
enum VehicleAPIError: LocalizedError {
    case notAuthenticated
    case authenticationFailed
    case commandFailed
    case networkError
    case invalidResponse
    case tokenExpired
    
    var errorDescription: String? {
        switch self {
        case .notAuthenticated:
```

```swift
@MainActor
```

```swift
    private init() {}
    
    // MARK: - Published Properties
    
    @Published var isAuthenticated: Bool = false
    @Published var currentVehicleId: String?

    private let baseURL = "https://api.toyota.com/vehicles"

    // MARK: - Authentication
    
    /// Authenticate with Toyota App
    /// - Parameters:
    ///   - email: User's Toyota App email
    ///   - password: User's Toyota App password
    /// - Throws: APIError on authentication failure
    func authenticate(email: String, password: String) async throws {
        let authURL = URL(string: "\(baseURL)/oauth/token")!
        
        var request = URLRequest(url: authURL)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        
        let body: [String: Any] = [
            "grant_type": "password",
            "client_id": clientID,
            "username": email,
            "password": password
        ]
        
        request.httpBody = try? JSONSerialization.data(withJSONObject: body)
        
        let (data, response) = try await URLSession.shared.data(for: request)
        
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            throw APIError.authenticationFailed
        }
        
        guard let json = try JSONSerialization.jsonObject(with: data) as? [String: Any],
              let accessToken = json["access_token"] as? String else {
            throw APIError.invalidResponse
        }
        
        // Store token in Keychain
        try KeychainHelper.saveToken(accessToken, for: "toyota")
        isAuthenticated = true
    }
    
    // MARK: - Climate Control
    
    /// Start climate control (remote start)
    /// - Throws: APIError on failure

        guard let token = try? KeychainHelper.getToken(for: "toyota"),

            throw APIError.notAuthenticated
        }
        
        let url = URL(string: "\(baseURL)/\(vehicleId)/remote/start")!

            throw APIError.commandFailed
        }

            throw APIError.commandFailed
        }
    }

    /// - Returns: ClimateState with current status
    /// - Throws: APIError on failure

        }
```

```swift
/// Sound Feedback Service
/// Plays sound notifications for warm-up completion
@MainActor
final class SoundFeedbackService {
    
    // MARK: - Singleton
    
    static let shared = SoundFeedbackService()
    private init() {}
    
    // MARK: - Audio Player
    
    private var audioPlayer: AVAudioPlayer?
    
    // MARK: - Sound Types
    
    enum SoundType: String, CaseIterable {
        case warmUpComplete = "warmup_complete"
        case timerStarted = "timer_started"
        case timerPaused = "timer_paused"
        case timerStopped = "timer_stopped"
        
        var fileName: String {
            return rawValue
        }
    }
    
    // MARK: - Play Sound
    
    /// Play sound notification
    /// - Parameters:
    ///   - soundType: The type of sound to play
    func playSound(_ soundType: SoundType) {
        guard let url = Bundle.main.url(forResource: soundType.fileName, withExtension: "mp3") else {
            print("Sound file not found: \(soundType.fileName)")
            return
        }
        
        do {
            audioPlayer = try AVAudioPlayer(contentsOf: url)
            audioPlayer?.play()
        } catch {
            print("Failed to play sound: \(error)")
        }
    }
    
    /// Stop currently playing sound
    func stopSound() {
        audioPlayer?.stop()
        audioPlayer = nil
    }
    
    // MARK: - Settings Integration
    
    /// Check if sound is enabled in settings
    /// - Returns: True if sound notifications are enabled
    var isSoundEnabled: Bool {
        return SettingsService.shared.settings.soundEnabled
    }
    
    /// Play sound if enabled
    /// - Parameters:
    ///   - soundType: The type of sound to play
    func playSoundIfEnabled(_ soundType: SoundType) {
        guard isSoundEnabled else { return }
        playSound(soundType)
    }
}
```
