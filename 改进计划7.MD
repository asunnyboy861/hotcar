## 📋 执行摘要

基于软件流程分析，当前软件已实现85%核心功能，但存在以下关键缺失：

1. **首次使用引导不足** - 新用户完成Onboarding后缺少车辆添加引导
2. **统计可视化缺失** - 有数据但无图表趋势展示
3. **社交分享未完成** - ShareService.generateShareImage() 返回 nil

本计划将针对以上三个核心问题进行优化，确保软件达到生产级可用性。

## 🎯 第一部分：首次使用引导优化

### 1.1 需求描述

**问题**: 用户完成Onboarding后直接进入主页面，如果没有车辆会显示"My Vehicle"占位符，但没有明确引导用户添加车辆。

**目标**: 
- 检测用户是否已添加车辆
- 未添加车辆时显示引导卡片
- 提供一键添加车辆入口
- 引导流程完成后自动刷新主页面

**用户流程**:
```
启动App → 检查Onboarding状态 → 检查车辆状态 → 
无车辆 → 显示引导卡片 → 点击添加 → 车辆列表 → 添加车辆 → 
返回主页 → 显示真实车辆数据
```

extension OnboardingManager {
    
    // MARK: - Vehicle Setup Guide State
    
    private var vehicleGuideShownKey: String { "com.hotcar.vehicleGuide.shown" }
    private var vehicleGuideDismissedKey: String { "com.hotcar.vehicleGuide.dismissed" }
    
    @Published var hasShownVehicleGuide: Bool
    @Published var hasDismissedVehicleGuide: Bool
    
    // 是否显示车辆引导
    var shouldShowVehicleGuide: Bool {
        hasCompletedOnboarding && 
        !hasAddedVehicle && 
        !hasDismissedVehicleGuide
    }
    
    func markVehicleGuideShown() {
        hasShownVehicleGuide = true
        UserDefaults.standard.set(true, forKey: vehicleGuideShownKey)
    }
    
    func dismissVehicleGuide() {
        hasDismissedVehicleGuide = true
        UserDefaults.standard.set(true, forKey: vehicleGuideDismissedKey)
    }
    
    func resetVehicleGuide() {
        hasShownVehicleGuide = false
        hasDismissedVehicleGuide = false
        UserDefaults.standard.removeObject(forKey: vehicleGuideShownKey)
        UserDefaults.standard.removeObject(forKey: vehicleGuideDismissedKey)
    }
}

// 单一职责：定义图表数据模型

import Foundation

/// Data point for time-based charts
/// Responsibility: Represent a single data point on time series charts
struct TimeSeriesDataPoint: Identifiable {
    let id = UUID()
    let date: Date
    let value: Double
    let label: String?
    
    init(date: Date, value: Double, label: String? = nil) {
        self.date = date
        self.value = value
        self.label = label
    }
}

/// Chart data container for warm-up statistics
/// Responsibility: Aggregate data points for chart display
struct WarmUpChartData {
    let timeRange: TimeRange
    let dataPoints: [TimeSeriesDataPoint]
    let totalValue: Double
    let averageValue: Double
    let maxValue: Double
    let minValue: Double
    
    /// Check if data is available for chart display
    var hasData: Bool {
        !dataPoints.isEmpty && dataPoints.contains { $0.value > 0 }
    }
}

/// Enum for chart types
enum ChartType: String, CaseIterable {
    case warmUpTime = "Warm-Up Time"
    case fuelConsumption = "Fuel Used"
    case sessionCount = "Sessions"
    
    var icon: String {
        switch self {
        case .warmUpTime: return "clock.fill"
        case .fuelConsumption: return "drop.fill"
        case .sessionCount: return "list.number"
        }
    }
    
    var unit: String {
        switch self {
        case .warmUpTime: return "min"
        case .fuelConsumption: return "L"
        case .sessionCount: return ""
        }
    }
}

// Features/Statistics/Components/WarmUpTrendChart.swift
// 单一职责：显示预热时间趋势图表

import SwiftUI
import Charts

/// Warm-up time trend chart component
/// Responsibility: Display time series chart of warm-up durations
struct WarmUpTrendChart: View {
    
    // MARK: - Properties
    
    let data: WarmUpChartData
    let chartType: ChartType
    @State private var selectedDataPoint: TimeSeriesDataPoint?
    
    // MARK: - Body
    
    var body: some View {
        VStack(alignment: .leading, spacing: HotCarSpacing.medium) {
            // Header
            chartHeader
            
            // Chart
            chartContent
                .frame(height: 200)
            
            // Selected value indicator
            if let selected = selectedDataPoint {
                selectedValueView(selected)
            }
        }
        .padding(HotCarSpacing.medium)
        .background(
            RoundedRectangle(cornerRadius: HotCarCornerRadius.medium)
                .fill(Color.backgroundSecondary)
        )
    }
    
    // MARK: - Chart Header
    
    private var chartHeader: some View {
        HStack {
            VStack(alignment: .leading, spacing: 4) {
                Text(chartType.rawValue)
                    .font(.hotCarHeadline)
                    .foregroundColor(.textPrimary)
                
                Text("Avg: \(String(format: "%.1f", data.averageValue)) \(chartType.unit)")
                    .font(.hotCarCaption)
                    .foregroundColor(.textSecondary)
            }
            
            Spacer()
            
            Image(systemName: chartType.icon)
                .font(.system(size: 24))
                .foregroundColor(.hotCarPrimary)
        }
    }
    
    // MARK: - Chart Content
    
    @ViewBuilder
    private var chartContent: some View {
        if data.hasData {
            Chart(data.dataPoints) { point in
                LineMark(
                    x: .value("Date", point.date),
                    y: .value("Value", point.value)
                )
                .foregroundStyle(Color.hotCarPrimary.gradient)
                .interpolationMethod(.catmullRom)
                
                AreaMark(
                    x: .value("Date", point.date),
                    y: .value("Value", point.value)
                )
                .foregroundStyle(Color.hotCarPrimary.opacity(0.1))
                .interpolationMethod(.catmullRom)
                
                if let selected = selectedDataPoint, selected.id == point.id {
                    RuleMark(x: .value("Date", point.date))
                        .foregroundStyle(Color.hotCarSecondary)
                        .lineStyle(StrokeStyle(lineWidth: 1, dash: [5]))
                }
            }
            .chartXAxis {
                AxisMarks(values: .automatic(desiredCount: 5)) { value in
                    AxisGridLine()
                    AxisValueLabel(format: .dateTime.month().day())
                }
            }
            .chartYAxis {
                AxisMarks { value in
                    AxisGridLine()
                    AxisValueLabel {
                        if let doubleValue = value.as(Double.self) {
                            Text("\(Int(doubleValue))")
                                .font(.hotCarCaption)
                        }
                    }
                }
            }
            .chartBackground { chartProxy in
                GeometryReader { geometry in
                    Rectangle()
                        .fill(Color.clear)
                        .contentShape(Rectangle())
                        .onTapGesture { location in
                            handleChartTap(location: location, geometry: geometry, chartProxy: chartProxy)
                        }
                }
            }
        } else {
            emptyChartView
        }
    }
    
    // MARK: - Empty Chart View
    
    private var emptyChartView: some View {
        VStack(spacing: 12) {
            Image(systemName: "chart.line.uptrend.xyaxis")
                .font(.system(size: 40))
                .foregroundColor(.textMuted)
            
            Text(NSLocalizedString("chart_no_data", tableName: "Localizable", comment: "Message when no chart data available"))
                .font(.hotCarBody)
                .foregroundColor(.textSecondary)
        }
        .frame(maxWidth: .infinity, maxHeight: .infinity)
    }
    
    // MARK: - Selected Value View
    
    private func selectedValueView(_ point: TimeSeriesDataPoint) -> some View {
        HStack {
            Text(point.date, style: .date)
                .font(.hotCarCaption)
                .foregroundColor(.textSecondary)
            
            Spacer()
            
            Text("\(String(format: "%.1f", point.value)) \(chartType.unit)")
                .font(.hotCarHeadline)
                .foregroundColor(.hotCarPrimary)
        }
        .padding(.horizontal, HotCarSpacing.small)
        .padding(.vertical, HotCarSpacing.small)
        .background(Color.hotCarPrimary.opacity(0.1))
        .cornerRadius(HotCarCornerRadius.small)
    }
    
    // MARK: - Helper Methods
    
    private func handleChartTap(location: CGPoint, geometry: GeometryProxy, chartProxy: ChartProxy) {
        // Convert tap location to chart coordinate
        let xPosition = location.x - geometry[chartProxy.plotAreaFrame].origin.x
        
        // Find nearest data point
        if let date: Date = chartProxy.value(atX: xPosition) {
            selectedDataPoint = data.dataPoints.min { point1, point2 in
                abs(point1.date.timeIntervalSince(date)) < abs(point2.date.timeIntervalSince(date))
            }
        }
    }
}

// Features/Statistics/StatisticsViewModel.swift
// 扩展：添加图表数据生成方法

extension StatisticsViewModel {
    
    // MARK: - Chart Data Generation
    
    /// Generate chart data for the current time range
    func generateChartData(for type: ChartType) -> WarmUpChartData {
        let sessions = getSessionsForCurrentRange()
        
        switch type {
        case .warmUpTime:
            return generateWarmUpTimeChartData(sessions: sessions)
        case .fuelConsumption:
            return generateFuelConsumptionChartData(sessions: sessions)
        case .sessionCount:
            return generateSessionCountChartData(sessions: sessions)
        }
    }
    
    private func generateWarmUpTimeChartData(sessions: [WarmUpSession]) -> WarmUpChartData {
        let groupedData = groupSessionsByDate(sessions)
        
        let dataPoints = groupedData.map { date, sessions in
            let avgDuration = sessions.map { Double($0.durationMinutes) }.reduce(0, +) / Double(sessions.count)
            return TimeSeriesDataPoint(date: date, value: avgDuration)
        }.sorted { $0.date < $1.date }
        
        let values = dataPoints.map { $0.value }
        
        return WarmUpChartData(
            timeRange: selectedTimeRange,
            dataPoints: dataPoints,
            totalValue: values.reduce(0, +),
            averageValue: values.isEmpty ? 0 : values.reduce(0, +) / Double(values.count),
            maxValue: values.max() ?? 0,
            minValue: values.min() ?? 0
        )
    }
    
    private func generateFuelConsumptionChartData(sessions: [WarmUpSession]) -> WarmUpChartData {
        let groupedData = groupSessionsByDate(sessions)
        
        let dataPoints = groupedData.map { date, sessions in
            let totalFuel = sessions.map { $0.fuelUsedLiters }.reduce(0, +)
            return TimeSeriesDataPoint(date: date, value: totalFuel)
        }.sorted { $0.date < $1.date }
        
        let values = dataPoints.map { $0.value }
        
        return WarmUpChartData(
            timeRange: selectedTimeRange,
            dataPoints: dataPoints,
            totalValue: values.reduce(0, +),
            averageValue: values.isEmpty ? 0 : values.reduce(0, +) / Double(values.count),
            maxValue: values.max() ?? 0,
            minValue: values.min() ?? 0
        )
    }
    
    private func groupSessionsByDate(_ sessions: [WarmUpSession]) -> [Date: [WarmUpSession]] {
        let calendar = Calendar.current
        var grouped: [Date: [WarmUpSession]] = [:]
        
        for session in sessions {
            let date = calendar.startOfDay(for: session.startTime)
            grouped[date, default: []].append(session)
        }
        
        return grouped
    }
}

import SwiftUI

/// Share view model
/// Responsibility: Manage share view state and handle sharing
@MainActor
final class ShareViewModel: ObservableObject {
    
    // MARK: - Published Properties
    
    @Published var selectedPeriod: ShareData.SharePeriod = .weekly
    @Published var shareData: ShareData
    @Published var isGeneratingImage = false
    @Published var generatedImage: UIImage?
    
    // MARK: - Dependencies
    
    private let shareService = ShareService.shared
    
    // MARK: - Initialization
    
    init() {
        shareData = shareService.generateShareData(for: .weekly)
    }
    
    // MARK: - Methods
    
    func updateShareData() {
        shareData = shareService.generateShareData(for: selectedPeriod)
    }
    
    func generateImage() {
        isGeneratingImage = true
        generatedImage = shareService.generateShareImage(from: shareData)
        isGeneratingImage = false
    }
    
    func share() {
        generateImage()
        
        guard let image = generatedImage else { return }
        
        // Present share sheet
        let activityItems: [Any] = [
            image,
            "Check out my HotCar stats! 🚗❄️"
        ]
        
        let activityViewController = UIActivityViewController(
            activityItems: activityItems,
            applicationActivities: nil
        )
        
        // Get root view controller
        if let windowScene = UIApplication.shared.connectedScenes.first as? UIWindowScene,
           let rootViewController = windowScene.windows.first?.rootViewController {
            rootViewController.present(activityViewController, animated: true)
        }
    }
}

    var displayName: String {
        switch self {
        case .weekly: return "This Week"
        case .monthly: return "This Month"
        case .allTime: return "All Time"
        }
    }
