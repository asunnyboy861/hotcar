## 📋 执行摘要

### 11.1 需求描述

**核心需求**：将现有的 Tesla API 远程启动功能集成到主页计时器，实现"一键预热"真正控制车辆空调系统。

**用户价值**：
- ✅ Tesla 车主无需手动启动车辆，App 直接控制空调
- ✅ 实时显示车内温度变化，可视化预热进度
- ✅ 自动停止计时器，无需手动取消
- ✅ 非 Tesla 车主继续使用纯计时器模式（功能不受影响）

**技术目标**：
1. 复用现有 `TeslaAPIService` 和 `RemoteStartViewModel`（已有完整实现）
2. 在 `HomeViewModel` 中添加混合模式支持（API 控制 + 纯计时器）
3. 在 `HomeView` 中显示车辆连接状态和实时温度
4. 保持向后兼容，无 API 连接时自动降级为纯计时器

## 🎯 设计原则

### 原则 1: 单一职责

每个模块只负责一个功能：
- `TeslaAPIService` - 仅负责 Tesla API 调用
- `RemoteStartViewModel` - 仅管理远程启动状态
- `HomeViewModel` - 仅协调计时器和远程启动
- `HomeView` - 仅负责 UI 展示

### 原则 2: 高内聚低耦合

模块间通过清晰的接口通信：
```
HomeViewModel → TeslaAPIService（依赖注入）
HomeViewModel → CountdownTimer（已有依赖）
HomeView → HomeViewModel（@Published 自动更新）
```

### 原则 3: 代码复用

优先复用现有模块：
- ✅ `TeslaAPIService.swift` - 已有完整 API 实现
- ✅ `RemoteStartViewModel.swift` - 已有认证和气候控制逻辑
- ✅ `CountdownTimer.swift` - 已有计时器功能
- ✅ `VehicleModel` - 已有 `apiToken` 字段

### 原则 4: 渐进式增强

- 有 API Token → 启用远程控制模式
- 无 API Token → 降级为纯计时器模式
- API 调用失败 → 自动降级，不影响基本功能

### 原则 5: 风格一致

遵循现有代码风格：
- SwiftUI + Combine 架构
- MVVM 模式
- 英文命名（符合北美市场）
- 深色主题 + 橙色强调色

## 📝 详细实现方案

### 模块 1: HomeViewModel 增强

**文件**: `HomeViewModel.swift`

**需求**: 添加 Tesla API 集成，支持混合模式（远程控制 + 纯计时器）

**生成规则**:

```swift
// MARK: - 新增属性

/// 车辆连接状态
@Published var vehicleConnectionState: VehicleConnectionState = .disconnected

/// 计时器模式
@Published var timerMode: TimerMode = .local

/// Tesla API 服务（复用现有单例）
private let teslaService = TeslaAPIService.shared

/// 是否正在监控空调状态
private var isMonitoringClimate = false

// MARK: - 新增枚举

enum VehicleConnectionState {
    case disconnected      // 无 API 连接
    case connecting        // 正在连接
    case connected         // 已连接
    case error(String)     // 连接错误
    
    var isConnected: Bool {
        if case .connected = self { return true }
        return false
    }
}

enum TimerMode {
    case local           // 纯本地计时
    case remote          // 远程控制模式
}

// MARK: - 修改 startTimer 方法

/// Start warm-up timer or remote climate control
/// - If vehicle has API token: Start remote climate control
/// - Otherwise: Start local countdown timer
func startTimer() {
    guard let vehicle = primaryVehicle else {
        showError("Please select a vehicle first")
        return
    }
    
    // Check if vehicle has API connection (Tesla or other OEM)
    if let apiToken = vehicle.apiToken, !apiToken.isEmpty {
        // Remote climate control mode
        Task {
            await startRemoteClimate()
        }
    } else {
        // Local timer mode (fallback)
        startLocalTimer()
    }
}

// MARK: - 新增方法

/// Start remote climate control via Tesla API with retry mechanism
@MainActor
private func startRemoteClimate() async {
    vehicleConnectionState = .connecting
    
    do {
        // 1. Start climate control with retry (max 3 attempts)
        try await startClimateWithRetry(maxRetries: 3)
        
        // 2. Update UI state
        isClimateOn = true
        isTimerActive = true
        timerMode = .remote
        vehicleConnectionState = .connected
        
        // 3. Start countdown timer
        countdownTimer.start(minutes: finalRecommendedTime)
        
        // 4. Start climate monitoring
        startClimateMonitoring()
        
        // 5. Update widget
        widgetService.updateClimateState(isOn: true, insideTemp: nil)
        
    } catch TeslaAPIService.APIError.networkError {
        // Network error: Auto-fallback to local timer
        vehicleConnectionState = .error("Network error")
        startLocalTimer()
    } catch TeslaAPIService.APIError.notAuthenticated {
        // Authentication expired: Show re-login prompt
        vehicleConnectionState = .error("Authentication expired")
        showError("Tesla authentication expired. Please re-login in Settings.")
        startLocalTimer()
    } catch {
        // Other errors: Fallback to local timer
        vehicleConnectionState = .error(error.localizedDescription)
        showError("Failed to start climate control. Starting local timer instead.")
        startLocalTimer()
    }
}

/// Start climate control with retry mechanism
/// - Parameter maxRetries: Maximum number of retry attempts
/// - Throws: APIError if all retries fail
private func startClimateWithRetry(maxRetries: Int = 3) async throws {
    var lastError: Error?
    
    for attempt in 1...maxRetries {
        do {
            try await teslaService.startClimate()
            return // Success
        } catch {
            lastError = error
            // Exponential backoff: wait 1s, 2s, 4s...
            let delay = UInt64(pow(2.0, Double(attempt - 1))) * 1_000_000_000
            try? await Task.sleep(nanoseconds: delay)
        }
    }
    
    // All retries failed
    throw lastError ?? TeslaAPIService.APIError.networkError
}

/// Start local countdown timer (no API connection)
/// Reuses existing timer logic as fallback
@MainActor
private func startLocalTimer() {
    timerMode = .local
    isTimerActive = true
    
    // Use current temperature or default (0°C) - don't block on weather failure
    let temperature = currentTemperature ?? 0
    let minutes = finalRecommendedTime
    
    countdownTimer.start(
        minutes: minutes,
        vehicleName: primaryVehicle?.name ?? "Vehicle",
        vehicleId: primaryVehicle?.id ?? "unknown",
        temperature: temperature
    )
    
    // Show notification for manual start
    showNotification(
        title: "Start Your Vehicle",
        body: "Please manually start your vehicle's engine or block heater"
    )
}

/// Start monitoring climate state with periodic polling
@MainActor
private func startClimateMonitoring() {
    guard !isMonitoringClimate else { return }
    isMonitoringClimate = true
    
    Task {
        while isClimateOn && timerMode == .remote {
            // Poll every 30 seconds
            try? await Task.sleep(nanoseconds: 30_000_000_000)
            await refreshClimateState()
        }
        isMonitoringClimate = false
    }
}

/// Refresh climate state from Tesla API
/// - Note: Converts temperature to user's preferred unit (°C/°F)
@MainActor
private func refreshClimateState() async {
    do {
        let climateState = try await teslaService.getClimateState()
        
        await MainActor.run {
            isClimateOn = climateState.isClimateOn
            
            // Convert temperature to user's preferred unit
            let tempUnit = settingsService.settings.temperatureUnit
            insideTemperature = tempUnit.convert(from: climateState.insideTemp ?? 0)
            outsideTemperature = tempUnit.convert(from: climateState.outsideTemp ?? 0)
            
            // Auto-stop timer if climate is off
            if !climateState.isClimateOn && isTimerActive {
                stopTimer()
            }
        }
    } catch {
        // Keep last known state on network error
        print("Failed to refresh climate state: \(error)")
    }
}

/// Stop timer and climate control
@MainActor
func stopTimer() {
    // Stop climate control if in remote mode
    if timerMode == .remote && isClimateOn {
        Task {
            try? await teslaService.stopClimate()
        }
    }
    
    // Stop countdown timer
    countdownTimer.stop()
    
    // Reset state
    isTimerActive = false
    isClimateOn = false
    isMonitoringClimate = false
    
    // Update widget
    widgetService.updateClimateState(isOn: false, insideTemp: nil)
}

// MARK: - 修改 adjustTimerTime 方法

/// Adjust timer time by adding/subtracting minutes
/// Note: In remote mode, this only adjusts the local countdown, not the actual climate control
func adjustTimerTime(by minutes: Int) {
    userTimeAdjustment += minutes
    
    // Ensure final time is at least 1 minute
    if finalRecommendedTime < 1 {
        userTimeAdjustment = 1 - calculatedWarmUpTime
    }
    
    // Update countdown timer if active
    if isTimerActive {
        let newMinutes = max(1, timeRemaining / 60 + minutes)
        countdownTimer.reset(minutes: newMinutes)
    }
    
    // Update widget with new recommended time
    widgetService.updateRecommendedTime(finalRecommendedTime)
}
```

**检验标准**:
- ✅ 有 API Token 时调用 `teslaService.startClimate()`
- ✅ API 调用失败时自动降级为本地计时器
- ✅ 每 30 秒轮询一次 Climate State
- ✅ 车内温度实时更新
- ✅ 空调关闭时自动停止计时器

// HomeViewModel.swift - 新增
@MainActor
private func startClimateWithRetry(maxRetries: Int = 3) async throws {
    var lastError: Error?
    
    for attempt in 1...maxRetries {
        do {
            try await teslaService.startClimate()
            return // Success
        } catch {
            lastError = error
            // Exponential backoff
            let delay = UInt64(pow(2.0, Double(attempt - 1))) * 1_000_000_000
            try? await Task.sleep(nanoseconds: delay)
        }
    }
    
    throw lastError ?? TeslaAPIService.APIError.networkError
}
