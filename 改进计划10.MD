# HotCar 改进计划 10 - 核心功能修复与稳定性优化

## 一、问题概述

当前软件存在严重的功能失效问题：
1. **点击"+/-"调整时间无响应** - 用户无法手动调整预热时间
2. **点击"Start Warm-up"无响应** - 核心功能完全失效
3. **天气API解码错误** - 导致温度数据获取失败
4. **图标兼容性问题** - 部分系统图标在旧版iOS中不存在

## 二、根本原因分析

### 2.1 天气API解码错误（关键）
- **问题**: `HourlyData.time` 定义为 `[Date]`，但Open-Meteo API返回ISO 8601字符串
- **影响**: `fetchTomorrowMorningTemperature()` 解码失败，`tomorrowTemperature`始终为nil
- **级联效应**: 可能间接影响`currentTemperature`获取

### 2.2 计时器调整逻辑缺陷
- **问题**: `adjustTimerTime` 使用`calculatedWarmUpTime`（计算属性），未保存用户调整值
- **影响**: 用户调整的时间无法持久化，UI不更新

### 2.3 Start Timer守卫条件过于严格
- **问题**: `startTimer()`要求`currentTemperature`必须非nil，否则静默返回
- **影响**: 天气获取失败时，核心功能完全不可用

### 2.4 图标名称不存在
- **问题**: `disc.fill`和`thermometer.liquid.fill`在iOS 15/16不存在
- **影响**: 可能导致视图渲染异常

## 三、优化目标

1. **修复天气API解码**，确保温度数据正确获取
2. **修复计时器调整功能**，支持用户手动调整时间
3. **优化Start Timer逻辑**，在天气获取失败时使用默认值
4. **修复图标兼容性**，替换为通用图标
5. **增强错误处理**，提供用户反馈而非静默失败

@MainActor
final class HomeViewModel: ObservableObject {
    // 新增：用户调整的时间偏移量
    @Published var userTimeAdjustment: Int = 0
    
    /// 计算最终推荐时间（基础计算 + 用户调整）

        if finalRecommendedTime < 1 {

        }
    }
    
    func resetTimeAdjustment() {
        userTimeAdjustment = 0
    }
}

### 5.1 天气解码修复
```swift
// MARK: - Weather Response Models

struct WeatherResponse: Codable {
    let currentWeather: CurrentWeather
    
    enum CodingKeys: String, CodingKey {
        case currentWeather = "current_weather"
    }
}

struct CurrentWeather: Codable {
    let temperature: Double
}

struct HourlyWeatherResponse: Codable {
    let hourly: HourlyData
}

struct HourlyData: Codable {
    let time: [String]  // ISO 8601 format strings
    let temperature: [Double]
    
    enum CodingKeys: String, CodingKey {
        case time
        case temperature = "temperature_2m"
    }
    
    /// Parse ISO 8601 strings to Date objects
    var parsedDates: [Date] {
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM-dd'T'HH:mm"
        formatter.timeZone = TimeZone(identifier: "UTC")
        
        return time.compactMap { formatter.date(from: $0) }
    }
}
```

### 5.2 获取明天温度更新
```swift
@MainActor
func fetchTomorrowMorningTemperature() async {
    let location = locationService.getCurrentLocation()
    
    let urlString = "\(baseURL)?latitude=\(location.latitude)&longitude=\(location.longitude)&hourly=temperature_2m&forecast_days=2&temperature_unit=celsius"
    
    guard let url = URL(string: urlString) else {
        return
    }
    
    do {
        let (data, response) = try await URLSession.shared.data(from: url)
        
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            return
        }
        
        let decoder = JSONDecoder()
        let weatherData = try decoder.decode(HourlyWeatherResponse.self, from: data)
        
        // Get temperature for 7 AM tomorrow using parsed dates
        let tomorrow7AM = Calendar.current.startOfDay(for: Date())
            .addingTimeInterval(24 * 60 * 60 + 7 * 60 * 60)
        
        let dates = weatherData.hourly.parsedDates
        if let index = dates.firstIndex(where: { $0 >= tomorrow7AM }) {
            await MainActor.run {
                self.tomorrowMorningTemp = weatherData.hourly.temperature[index]
            }
        }
    } catch {
        print("Error fetching tomorrow's temperature: \(error)")
    }
}
```

@MainActor
final class HomeViewModel: ObservableObject {
    
    // MARK: - Published Properties
    
    @Published var currentTemperature: Double?
    @Published var tomorrowTemperature: Double?
    @Published var locationName: String = NSLocalizedString("loading", tableName: "Localizable", comment: "Loading state text")
    @Published var isLoading: Bool = false
    
    @Published var isTimerActive: Bool = false
    @Published var isTimerPaused: Bool = false
    @Published var timeRemaining: Int = 0
    @Published var timerProgress: Double = 0.0
    
    // MARK: - User Adjustment
    
    /// User manual time adjustment in minutes
    @Published var userTimeAdjustment: Int = 0
    
    /// Error handling
    @Published var errorMessage: String?
    @Published var showErrorAlert: Bool = false
    
    // ... 其他属性 ...
    
    // MARK: - Computed Properties
    
    /// Base calculated warm-up time
    var calculatedWarmUpTime: Int {
        guard let temperature = currentTemperature,
              let vehicle = primaryVehicle else {
            return settingsService.settings.defaultTimerDuration
        }
        
        return WarmUpCalculationEngine.calculateWarmUpTime(
            temperature: temperature,
            vehicleType: vehicle.type,
            engineType: vehicle.engineType,
            hasBlockHeater: vehicle.hasBlockHeater,
            isPluggedIn: vehicle.isPluggedIn
        )
    }
    
    /// Final recommended time including user adjustment
    var finalRecommendedTime: Int {
        let baseTime = calculatedWarmUpTime
        return max(1, baseTime + userTimeAdjustment)
    }
    
    // MARK: - Timer Control
    
    func startTimer() {
        guard let vehicle = primaryVehicle else {
            showError("Please select a vehicle first")
            return
        }
        
        // Use current temperature or default (0°C)
        let temperature = currentTemperature ?? 0
        let minutes = finalRecommendedTime
        
        countdownTimer.start(
            minutes: minutes,
            vehicleName: vehicle.name,
            vehicleId: vehicle.id,
            temperature: temperature
        )
    }
    
    func adjustTimerTime(by minutes: Int) {
        userTimeAdjustment += minutes
        
        // Ensure minimum 1 minute
        if finalRecommendedTime < 1 {
            userTimeAdjustment = 1 - calculatedWarmUpTime
        }
    }
    
    func resetTimeAdjustment() {
        userTimeAdjustment = 0
    }
    
    // MARK: - Error Handling
    
    func showError(_ message: String) {
        errorMessage = message
        showErrorAlert = true
    }
}
