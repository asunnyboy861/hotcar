## 一、重构原则与规范（保持不变）

### 1.1 六大核心原则

| 原则 | 说明 | 检验标准 |
|------|------|----------|
| **单一职责** | 每个模块只负责一个功能 | 一个文件只处理一类逻辑 |
| **高内聚低耦合** | 相关功能聚合，模块间依赖最小化 | 模块间通过协议/接口通信 |
| **命名清晰** | 语义化命名，自解释代码 | 命名无需注释即可理解 |
| **代码复用** | 优先复用现有模块，遵循"三次法则" | 重复代码出现3次才抽象 |
| **最小代码** | 删除废弃代码，保持仓库整洁 | 无冗余文件和无引用代码 |
| **风格一致** | 与现有UI/功能风格保持一致 | 视觉和交互风格统一 |

### 4.1 优化1：温度状态判断（替代 TemperatureUnitManager）

#### 需求描述
- ❌ 不创建新的 TemperatureUnitManager
- ✅ 复用现有的 `AppSettings.TemperatureUnit`
- ✅ 添加 `TemperatureStatus` 枚举到现有代码

#### 生成规则（已修复）

```swift
// MARK: - TemperatureStatus.swift
// 文件位置: Features/Temperature/TemperatureStatus.swift
// ✅ 修复：只添加状态判断，不复用单位转换

import SwiftUI

/// 温度状态枚举
/// 职责：统一温度状态判断（内部统一使用摄氏度）
enum TemperatureStatus: CaseIterable {
    case extreme      // < -30°C
    case veryCold     // -30°C to -20°C
    case cold         // -20°C to -10°C
    case mild         // -10°C to 0°C
    case aboveFreezing // > 0°C
    
    /// 从摄氏度判断状态
    static func from(celsius: Double) -> TemperatureStatus {
        switch celsius {
        case ..<(-30): return .extreme
        case -30..<(-20): return .veryCold
        case -20..<(-10): return .cold
        case -10..<0: return .mild
        default: return .aboveFreezing
        }
    }
    
    /// 显示文本
    var displayText: String {
        switch self {
        case .extreme: return "Extreme Cold Alert"
        case .veryCold: return "Very Cold Conditions"
        case .cold: return "Cold Weather"
        case .mild: return "Below Freezing"
        case .aboveFreezing: return "Above Freezing"
        }
    }
    
    /// 对应颜色
    var color: Color {
        switch self {
        case .extreme: return .tempExtreme
        case .veryCold: return .tempVeryCold
        case .cold: return .tempCold
        case .mild: return .tempMild
        case .aboveFreezing: return .hotCarPrimary
        }
    }
    
    /// 图标
    var icon: String {
        switch self {
        case .extreme: return "exclamationmark.triangle.fill"
        case .veryCold: return "snowflake"
        case .cold: return "thermometer.snowflake"
        case .mild: return "thermometer.low"
        case .aboveFreezing: return "thermometer.medium"
        }
    }
}

// MARK: - AppSettings.TemperatureUnit 扩展示例

extension AppSettings.TemperatureUnit {
    /// 获取温度状态（辅助方法）
    func status(for celsius: Double) -> TemperatureStatus {
        TemperatureStatus.from(celsius: celsius)
    }
}
```

```swift
                    Text("\(recommendedMinutes)")
                        .font(.hotCarDisplay)
                        .foregroundColor(.textPrimary)

                        .font(.hotCarTitle)
                        .foregroundColor(.textSecondary)

                    .font(.hotCarCaption)
                    .foregroundColor(.textMuted)
            }
```

### 修复说明
- 使用 `@StateObject` 正确管理依赖
- 复用现有的 `SettingsService.shared`
- 正确处理车辆选择（使用 Primary Vehicle）

```swift
// MARK: - HomeViewModel.swift（修复版）
// 文件位置: Features/Home/HomeViewModel.swift

import Foundation
import Combine

@MainActor
final class HomeViewModel: ObservableObject {
    
    // MARK: - Published Properties
    
    @Published var currentTemperature: Double?
    @Published var tomorrowTemperature: Double?
    @Published var locationName: String = "Loading..."
    @Published var isLoading: Bool = false
    @Published var isTimerActive: Bool = false
    @Published var isTimerPaused: Bool = false
    @Published var timeRemaining: Int = 0
    @Published var timerProgress: Double = 0.0
    
    // MARK: - Dependencies（使用 @StateObject 管理）
    
    @Published var settingsService = SettingsService.shared  // ✅ 使用 @Published 监听变化
    @Published var vehicleService = VehicleService.shared
    @Published var statisticsService = StatisticsService.shared
    
    private let weatherService = WeatherService.shared
    private let countdownTimer = CountdownTimer.shared
    private var cancellables = Set<AnyCancellable>()
    
    // MARK: - Computed Properties
    
    /// ✅ 修复：从 settingsService 获取温度单位
    var temperatureUnit: AppSettings.TemperatureUnit {
        settingsService.settings.temperatureUnit
    }
    
    /// ✅ 修复：使用 getPrimaryVehicle() 获取主车辆
    var primaryVehicle: Vehicle? {
        vehicleService.getPrimaryVehicle()
    }
    
    /// ✅ 修复：使用 TemperatureStatus 判断状态
    var temperatureStatus: TemperatureStatus? {
        guard let temp = currentTemperature else { return nil }
        return TemperatureStatus.from(celsius: temp)
    }
    
    var calculatedWarmUpTime: Int {
        guard let temperature = currentTemperature,
              let vehicle = primaryVehicle else {
            return settingsService.settings.defaultTimerDuration
        }
        
        return WarmUpCalculationEngine.calculateWarmUpTime(
            temperature: temperature,
            vehicleType: vehicle.type,
            engineType: vehicle.engineType,
            hasBlockHeater: vehicle.hasBlockHeater,
            isPluggedIn: vehicle.isPluggedIn
        )
    }
    
    var calculatedWarmUpTimeForTomorrow: Int {
        guard let temperature = tomorrowTemperature,
              let vehicle = primaryVehicle else {
            return settingsService.settings.defaultTimerDuration
        }
        
        return WarmUpCalculationEngine.calculateWarmUpTime(
            temperature: temperature,
            vehicleType: vehicle.type,
            engineType: vehicle.engineType,
            hasBlockHeater: vehicle.hasBlockHeater,
            isPluggedIn: vehicle.isPluggedIn
        )
    }
    
    // MARK: - Initialization
    
    init() {
        setupBindings()
        loadInitialData()
    }
    
    // MARK: - Setup
    
    private func setupBindings() {
        // 监听计时器状态
        countdownTimer.$timeRemaining
            .receive(on: RunLoop.main)
            .sink { [weak self] time in
                self?.timeRemaining = time
                self?.timerProgress = self?.countdownTimer.progress ?? 0
            }
            .store(in: &cancellables)
        
        countdownTimer.$isPaused
            .receive(on: RunLoop.main)
            .sink { [weak self] isPaused in
                self?.isTimerPaused = isPaused
            }
            .store(in: &cancellables)
        
        countdownTimer.$isRunning
            .receive(on: RunLoop.main)
            .sink { [weak self] isRunning in
                self?.isTimerActive = isRunning
            }
            .store(in: &cancellables)
    }
    
    // MARK: - Public Methods
    
    func loadInitialData() {
        Task {
            await loadWeather()
            await vehicleService.loadVehicles()
        }
    }
    
    func loadWeather() async {
        isLoading = true
        
        await weatherService.fetchCurrentTemperature()
        await weatherService.fetchTomorrowMorningTemperature()
        
        self.currentTemperature = weatherService.currentTemperature
        self.tomorrowTemperature = weatherService.tomorrowMorningTemp
        self.locationName = "Current Location"  // TODO: 从 LocationService 获取
        
        isLoading = false
    }
    
    // MARK: - Timer Control
    
    func startTimer() {
        guard let vehicle = primaryVehicle,
              let temperature = currentTemperature else {
            return
        }
        
        let minutes = calculatedWarmUpTime
        
        countdownTimer.start(
            minutes: minutes,
            vehicleName: vehicle.name,
            vehicleId: vehicle.id,
            temperature: temperature
        )
    }
    
    func stopTimer() {
        countdownTimer.stop()
    }
    
    func adjustTimerTime(by minutes: Int) {
        let newMinutes = max(1, calculatedWarmUpTime + minutes)
        countdownTimer.reset(minutes: newMinutes)
    }
    
    // MARK: - Vehicle Control（✅ 修复：使用现有方法）
    
    func setPrimaryVehicle(_ vehicle: Vehicle) {
        Task {
            await vehicleService.setPrimaryVehicle(vehicle)
        }
    }
}
```
