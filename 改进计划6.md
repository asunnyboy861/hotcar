## 一、重构原则与规范

### 1.1 六大核心原则

| 原则 | 说明 | 检验标准 |
|------|------|----------|
| **单一职责** | 每个模块只负责一个功能 | 一个文件只处理一类逻辑 |
| **高内聚低耦合** | 相关功能聚合，模块间依赖最小化 | 模块间通过协议/接口通信 |
| **命名清晰** | 语义化命名，自解释代码 | 命名无需注释即可理解 |
| **代码复用** | 优先复用现有模块，遵循"三次法则" | 重复代码出现3次才抽象 |
| **最小代码** | 删除废弃代码，保持仓库整洁 | 无冗余文件和无引用代码 |
| **风格一致** | 与现有UI/功能风格保持一致 | 视觉和交互风格统一 |

## 四、详细优化方案

### 4.1 优化1：统一温度单位系统

#### 需求描述
- 内部统一使用摄氏度（Celsius）进行所有计算
- 显示层根据用户Settings自动转换
- 温度状态判断（寒冷等级）统一使用摄氏度阈值

#### 生成规则

```swift
// MARK: - TemperatureUnitManager.swift
// 文件位置: Features/Temperature/TemperatureUnitManager.swift

import Foundation

/// 温度单位管理器
/// 职责：统一处理温度单位转换和显示
final class TemperatureUnitManager {
    
    // MARK: - Singleton
    static let shared = TemperatureUnitManager()
    private init() {}
    
    // MARK: - 转换方法
    
    /// 摄氏度转华氏度
    func toFahrenheit(_ celsius: Double) -> Double {
        return (celsius * 9/5) + 32
    }
    
    /// 华氏度转摄氏度
    func toCelsius(_ fahrenheit: Double) -> Double {
        return (fahrenheit - 32) * 5/9
    }
    
    /// 根据设置单位格式化显示
    func format(_ celsius: Double, unit: AppSettings.TemperatureUnit) -> String {
        let value = unit.convert(from: celsius)
        return String(format: "%.0f%@", value, unit.symbol)
    }
    
    // MARK: - 温度状态判断（统一使用摄氏度）
    
    enum TemperatureStatus {
        case extreme      // < -30°C
        case veryCold     // -30°C to -20°C
        case cold         // -20°C to -10°C
        case mild         // -10°C to 0°C
        case aboveFreezing // > 0°C
        
        static func from(celsius: Double) -> TemperatureStatus {
            switch celsius {
            case ..<(-30): return .extreme
            case -30..<(-20): return .veryCold
            case -20..<(-10): return .cold
            case -10..<0: return .mild
            default: return .aboveFreezing
            }
        }
        
        var displayText: String {
            switch self {
            case .extreme: return "Extreme Cold Alert"
            case .veryCold: return "Very Cold Conditions"
            case .cold: return "Cold Weather"
            case .mild: return "Below Freezing"
            case .aboveFreezing: return "Above Freezing"
            }
        }
        
        var color: Color {
            switch self {
            case .extreme: return .tempExtreme
            case .veryCold: return .tempVeryCold
            case .cold: return .tempCold
            case .mild: return .tempMild
            case .aboveFreezing: return .hotCarPrimary
            }
        }
    }
}
```

### 4.2 优化2：重构HeroTimerCard（核心计时器卡片）

#### 需求描述
- 将"建议时间显示"和"计时器控制"分离
- 大字体显示建议预热时间
- 一键开始默认时间
- 提供 +/- 调整（始终可见）

#### 生成规则

```swift
// MARK: - HeroTimerCard.swift
// 文件位置: Features/Home/Components/HeroTimerCard.swift

import SwiftUI

/// 核心计时器卡片
/// 职责：显示建议预热时间并提供开始/停止控制
struct HeroTimerCard: View {
    
    // MARK: - Properties
    
    let recommendedMinutes: Int
    let isRunning: Bool
    let progress: Double
    let remainingSeconds: Int
    let vehicleName: String
    let temperatureStatus: TemperatureUnitManager.TemperatureStatus
    
    let onStart: () -> Void
    let onStop: () -> Void
    let onAdjustTime: (Int) -> Void
    
    // MARK: - Body
    
    var body: some View {
        VStack(spacing: HotCarSpacing.large) {
            // 顶部：状态指示
            statusHeader
            
            // 中部：时间显示（核心）
            timeDisplay
            
            // 底部：控制区域
            controlSection
        }
        .padding(HotCarSpacing.large)
        .background(cardBackground)
        .cornerRadius(HotCarRadius.maximum)
        .shadow(color: shadowColor, radius: 20, x: 0, y: 10)
    }
    
    // MARK: - Status Header
    
    private var statusHeader: some View {
        HStack {
            Image(systemName: temperatureStatus.icon)
                .foregroundColor(temperatureStatus.color)
            
            Text(temperatureStatus.displayText)
                .font(.hotCarCaption)
                .foregroundColor(temperatureStatus.color)
            
            Spacer()
            
            Text(vehicleName)
                .font(.hotCarCaption)
                .foregroundColor(.textMuted)
        }
    }
    
    // MARK: - Time Display
    
    private var timeDisplay: some View {
        VStack(spacing: HotCarSpacing.small) {
            if isRunning {
                // 倒计时模式
                Text(formattedRemainingTime)
                    .font(.hotCarTimer)
                    .foregroundColor(.textPrimary)
                
                // 进度条
                ProgressView(value: progress)
                    .progressViewStyle(LinearProgressViewStyle(tint: .hotCarPrimary))
                    .frame(height: 4)
            } else {
                // 建议时间模式
                HStack(alignment: .firstTextBaseline, spacing: 4) {
                    Text("\(recommendedMinutes)")
                        .font(.hotCarDisplay)
                        .foregroundColor(.textPrimary)
                    
                    Text("min")
                        .font(.hotCarTitle)
                        .foregroundColor(.textSecondary)
                }
                
                Text("Recommended Warm-Up")
                    .font(.hotCarCaption)
                    .foregroundColor(.textMuted)
            }
        }
    }
    
    private var formattedRemainingTime: String {
        let mins = remainingSeconds / 60
        let secs = remainingSeconds % 60
        return String(format: "%02d:%02d", mins, secs)
    }
    
    // MARK: - Control Section
    
    private var controlSection: some View {
        VStack(spacing: HotCarSpacing.medium) {
            // 主按钮
            Button(action: isRunning ? onStop : onStart) {
                HStack {
                    Image(systemName: isRunning ? "stop.fill" : "play.fill")
                    Text(isRunning ? "Stop" : "Start Warm-Up")
                }
                .font(.hotCarButton)
                .foregroundColor(.white)
                .frame(maxWidth: .infinity)
                .padding(.vertical, HotCarSpacing.medium)
                .background(isRunning ? Color.warmUpActive : Color.hotCarPrimary)
                .cornerRadius(HotCarRadius.large)
            }
            .buttonStyle(PlainButtonStyle())
            
            // 时间调整（始终可见）
            HStack(spacing: HotCarSpacing.large) {
                Button(action: { onAdjustTime(-1) }) {
                    Image(systemName: "minus.circle.fill")
                        .font(.system(size: 32))
                        .foregroundColor(.textSecondary)
                }
                
                Text("\(recommendedMinutes) min")
                    .font(.hotCarHeadline)
                    .foregroundColor(.textSecondary)
                    .frame(minWidth: 80)
                
                Button(action: { onAdjustTime(1) }) {
                    Image(systemName: "plus.circle.fill")
                        .font(.system(size: 32))
                        .foregroundColor(.textSecondary)
                }
            }
        }
    }
    
    // MARK: - Styling
    
    private var cardBackground: some View {
        LinearGradient(
            gradient: Gradient(colors: [
                Color.backgroundSecondary,
                Color.backgroundCard
            ]),
            startPoint: .top,
            endPoint: .bottom
        )
    }
    
    private var shadowColor: Color {
        isRunning ? Color.warmUpActive.opacity(0.3) : Color.black.opacity(0.3)
    }
}
```
