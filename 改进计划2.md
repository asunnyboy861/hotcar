## 🎯 优化原则

### 1. 功能模块原则
- ✅ **单一职责**: 每个模块只负责一个功能
- ✅ **高内聚低耦合**: 模块内部紧密关联，模块之间松散耦合
- ✅ **命名清晰**: 使用语义化命名，见名知意
- ✅ **文件结构语义化**: 目录结构清晰反映功能划分

### 2. 代码复用原则
- ✅ **优先复用现有模块**: 不重复造轮子
- ✅ **三次法则**: 遇到三次重复代码再抽象
- ✅ **最小代码原则**: 能合并的尽量合并

### 3. 重构清理原则
- ✅ **标记废弃**: 使用 `@available(*, deprecated)` 标记
- ✅ **验证无影响**: 确保无引用后再删除
- ✅ **提交信息清晰**: 注明清理范围和原因

### 4. 风格一致性原则
- ✅ **UI 风格统一**: 遵循现有设计系统（ColorPalette, Typography, Spacing）
- ✅ **代码风格一致**: 遵循项目编码规范（注释、命名、结构）
- ✅ **功能逻辑连贯**: 与现有功能无缝衔接

### 5. 继承兼容原则
- ✅ **优先继承**: 利用现有类和协议
- ✅ **避免硬编码**: 使用配置和继承
- ✅ **向后兼容**: 不破坏现有功能
- ✅ **Settings 集成**: 所有可配置项必须集成到 AppSettings

```swift
import Foundation
import Combine

@MainActor
final class WeatherAlertService: ObservableObject {
    
    // MARK: - Singleton
    
    static let shared = WeatherAlertService()
    
    // MARK: - Published Properties
    
    @Published var activeAlerts: [WeatherAlert] = []
    @Published var isMonitoring: Bool = false
    
    // MARK: - Dependencies
    
    private let weatherService = WeatherService.shared
    private let notificationService = NotificationService.shared
    private let settingsService = SettingsService.shared
    private let storageKey = "com.hotcar.weatheralerts"
    private var cancellables = Set<AnyCancellable>()
    
    // MARK: - Initialization
    
    private init() {
        loadAlertsSync()
        setupAutoMonitoring()
    }
    
    // MARK: - Public Methods
    
    func loadAlerts() async {
        loadAlertsSync()
    }
    
    private func loadAlertsSync() {
        guard let data = UserDefaults.standard.data(forKey: storageKey),
              let decoded = try? JSONDecoder().decode([WeatherAlert].self, from: data) else {
            activeAlerts = []
            return
        }
        activeAlerts = decoded.filter { $0.expiresAt == nil || $0.expiresAt! > Date() }
    }
    
    /// Check current weather conditions and generate alerts
    func checkAndGenerateAlerts() async {
        guard let currentTemp = weatherService.currentTemperature else { return }
        
        var newAlerts: [WeatherAlert] = []
        
        // Check extreme cold
        if let alert = checkExtremeCold(temperature: currentTemp) {
            newAlerts.append(alert)
        }
        
        // Check temperature drop
        if let alert = checkTemperatureDrop() {
            newAlerts.append(alert)
        }
        
        // Save and notify
        for alert in newAlerts {
            await saveAlert(alert)
            await notificationService.sendWeatherAlert(alert)
        }
    }
    
    /// Mark alert as read
    func markAlertAsRead(_ alertId: String) async {
        if let index = activeAlerts.firstIndex(where: { $0.id == alertId }) {
            activeAlerts[index].isRead = true
            saveAlertsSync()
        }
    }
    
    /// Clear all alerts
    func clearAllAlerts() async {
        activeAlerts.removeAll()
        saveAlertsSync()
    }
    
    // MARK: - Private Methods
    
    private func checkExtremeCold(temperature: Double) -> WeatherAlert? {
        guard temperature < -30 else { return nil }
        
        return WeatherAlert(
            id: UUID().uuidString,
            type: .extremeCold,
            title: "⚠️ Extreme Cold Warning",
            message: String(format: "Current temperature is %.1f°C. Use block heater and extend warm-up time.", temperature),
            createdAt: Date(),
            expiresAt: nil,
            isRead: false,
            metadata: AlertMetadata(currentTemperature: temperature)
        )
    }
    
    private func checkTemperatureDrop() -> WeatherAlert? {
        // TODO: Implement with historical data
        return nil
    }
    
    private func saveAlert(_ alert: WeatherAlert) async {
        activeAlerts.append(alert)
        saveAlertsSync()
    }
    
    private func saveAlertsSync() {
        if let encoded = try? JSONEncoder().encode(activeAlerts) {
            UserDefaults.standard.set(encoded, forKey: storageKey)
        }
    }
    
    private func setupAutoMonitoring() {
        // Monitor weather changes
        weatherService.$currentTemperature
            .receive(on: RunLoop.main)
            .sink { [weak self] _ in
                Task {
                    await self?.checkAndGenerateAlerts()
                }
            }
            .store(in: &cancellables)
    }
}
```

**SmartReminderService.swift（新增）**
```swift
//
//  SmartReminderService.swift
//  hotcar
//
//  HotCar Service - Smart Reminder Management
//  Manages user departure schedules and triggers reminders
//

import Foundation
import Combine
import UserNotifications

@MainActor
final class SmartReminderService: ObservableObject {
    
    // MARK: - Singleton
    
    static let shared = SmartReminderService()
    
    // MARK: - Published Properties
    
    @Published var schedules: [ReminderSchedule] = []
    @Published var isMonitoring: Bool = false
    
    // MARK: - Dependencies
    
    private let notificationService = NotificationService.shared
    private let vehicleService = VehicleService.shared
    private let weatherService = WeatherService.shared
    private let warmUpCalculator = WarmUpCalculationEngine.self
    private let storageKey = "com.hotcar.smartreminders"
    private var cancellables = Set<AnyCancellable>()
    
    // MARK: - Initialization
    
    private init() {
        loadSchedulesSync()
        setupMonitoring()
    }
    
    // MARK: - Public Methods
    
    func loadSchedules() async {
        loadSchedulesSync()
    }
    
    private func loadSchedulesSync() {
        guard let data = UserDefaults.standard.data(forKey: storageKey),
              let decoded = try? JSONDecoder().decode([ReminderSchedule].self, from: data) else {
            schedules = []
            return
        }
        schedules = decoded
    }
    
    /// Add or update reminder schedule
    func saveSchedule(_ schedule: ReminderSchedule) async {
        if let index = schedules.firstIndex(where: { $0.id == schedule.id }) {
            schedules[index] = schedule
        } else {
            schedules.append(schedule)
        }
        saveSchedulesSync()
        await scheduleNotification(for: schedule)
    }
    
    /// Delete schedule
    func deleteSchedule(_ scheduleId: String) async {
        schedules.removeAll { $0.id == scheduleId }
        saveSchedulesSync()
        await cancelNotification(for: scheduleId)
    }
    
    /// Toggle schedule enabled
    func toggleSchedule(_ scheduleId: String) async {
        if let index = schedules.firstIndex(where: { $0.id == scheduleId }) {
            schedules[index].isEnabled.toggle()
            saveSchedulesSync()
            
            if schedules[index].isEnabled {
                await scheduleNotification(for: schedules[index])
            } else {
                await cancelNotification(for: scheduleId)
            }
        }
    }
    
    // MARK: - Private Methods
    
    private func saveSchedulesSync() {
        if let encoded = try? JSONEncoder().encode(schedules) {
            UserDefaults.standard.set(encoded, forKey: storageKey)
        }
    }
    
    private func scheduleNotification(for schedule: ReminderSchedule) async {
        guard schedule.isEnabled else { return }
        
        let content = UNMutableNotificationContent()
        content.title = "⏰ Time to Warm Up Your Car"
        content.body = "Departure at \(formatTime(schedule.departureTime))"
        content.sound = .default
        content.categoryIdentifier = "SMART_REMINDER"
        
        // Calculate trigger time (departure time - warm-up time - advance notice)
        let warmUpTime = schedule.customWarmUpTime ?? calculateWarmUpTime()
        let triggerDate = schedule.departureTime
            .addingTimeInterval(-TimeInterval(warmUpTime * 60))
            .addingTimeInterval(-TimeInterval(schedule.advanceNoticeMinutes * 60))
        
        let dateComponents = Calendar.current.dateComponents(
            [.hour, .minute],
            from: triggerDate
        )
        
        let trigger = UNCalendarNotificationTrigger(
            dateMatching: dateComponents,
            repeats: true
        )
        
        let request = UNNotificationRequest(
            identifier: "smart_reminder_\(schedule.id)",
            content: content,
            trigger: trigger
        )
        
        try? await UNUserNotificationCenter.current().add(request)
    }
    
    private func cancelNotification(for scheduleId: String) async {
        UNUserNotificationCenter.current()
            .removePendingNotificationRequests(withIdentifiers: ["smart_reminder_\(scheduleId)"])
    }
    
    private func calculateWarmUpTime() -> Int {
        guard let temp = weatherService.currentTemperature,
              let vehicle = vehicleService.primaryVehicle else {
            return 10 // Default
        }
        
        return warmUpCalculator.calculateWarmUpTime(
            temperature: temp,
            vehicleType: vehicle.type,
            engineType: vehicle.engineType,
            hasBlockHeater: vehicle.hasBlockHeater,
            isPluggedIn: vehicle.isPluggedIn
        )
    }
    
    private func formatTime(_ date: Date) -> String {
        let formatter = DateFormatter()
        formatter.timeStyle = .short
        return formatter.string(from: date)
    }
    
    private func setupMonitoring() {
        // Monitor weather changes to update warm-up time calculations
        weatherService.$currentTemperature
            .receive(on: RunLoop.main)
            .sink { [weak self] _ in
                Task {
                    await self?.rescheduleAllNotifications()
                }
            }
            .store(in: &cancellables)
    }
    
    private func rescheduleAllNotifications() async {
        for schedule in schedules where schedule.isEnabled {
            await cancelNotification(for: schedule.id)
            await scheduleNotification(for: schedule)
        }
    }
}
```

```swift
    // MARK: - Published Properties
    
    @Published var isMonitoring: Bool = false
    @Published var idleDuration: Int = 0 // seconds
    @Published var warningLevel: IdleWarningLevel = .none
    
    // MARK: - Dependencies
    
    private let settingsService = SettingsService.shared
    private let notificationService = NotificationService.shared
    private let statisticsService = StatisticsService.shared
    private var idleTimer: Timer?
    private var currentSessionId: String?
    
    // MARK: - Initialization
    
    private init() {}
    
    // MARK: - Public Methods
    
    /// Start monitoring after timer completion
    func startMonitoring(sessionId: String) {
        stopMonitoring()
        
        isMonitoring = true
        idleDuration = 0
        warningLevel = .none
        currentSessionId = sessionId
        
        idleTimer = Timer.scheduledTimer(withTimeInterval: 1.0, repeats: true) { [weak self] _ in
            Task { @MainActor in
                self?.tick()
            }
        }
    }
    
    /// Stop monitoring
    func stopMonitoring() {
        idleTimer?.invalidate()
        idleTimer = nil
        isMonitoring = false
        idleDuration = 0
        warningLevel = .none
        currentSessionId = nil
    }
    
    /// Mark engine as stopped by user
    func markEngineStopped() async {
        guard let sessionId = currentSessionId else { return }
        
        // Record idle waste statistics
        if idleDuration > 0 {
            await statisticsService.recordIdleWaste(
                sessionId: sessionId,
                idleSeconds: idleDuration
            )
        }
        
        stopMonitoring()
    }
    
    // MARK: - Private Methods
    
    private func tick() {
        idleDuration += 1
        
        let warningThresholds = settingsService.settings.idleWarningThresholds
        
        switch idleDuration {
        case warningThresholds.first:
            warningLevel = .level1
            Task {
                await notificationService.sendIdleWarning(level: .level1, duration: idleDuration)
            }
        case warningThresholds[1]:
            warningLevel = .level2
            Task {
                await notificationService.sendIdleWarning(level: .level2, duration: idleDuration)
            }
        case warningThresholds[2]:
            warningLevel = .level3
            Task {
                await notificationService.sendIdleWarning(level: .level3, duration: idleDuration)
            }
        default:
            break
        }
    }
}

// MARK: - Warning Levels

enum IdleWarningLevel: Int {
    case none = 0
    case level1 = 1  // 5 minutes
    case level2 = 2  // 10 minutes
    case level3 = 3  // 15 minutes
    
    var message: String {
        switch self {
        case .none: return ""
        case .level1: return "Engine has been idling for 5 minutes"
        case .level2: return "Engine has been idling for 10 minutes. Consider turning it off."
        case .level3: return "Engine has been idling for 15 minutes. High fuel waste!"
        }
    }
}

// MARK: - AppSettings Extension

extension AppSettings {
    var idleWarningThresholds: [Int] {
        [300, 600, 900] // 5, 10, 15 minutes in seconds
    }
}
```

```swift
//  HotCar Service - Geofence Management

import Foundation
import CoreLocation

@MainActor
final class GeoFenceService: NSObject, ObservableObject {
    
    // MARK: - Singleton
    
    static let shared = GeoFenceService()
    
    // MARK: - Published Properties
    
    @Published var isInsideGarage: Bool = false
    @Published var savedLocations: [SavedLocation] = []
    
    // MARK: - Dependencies
    
    private let locationManager = CLLocationManager()
    private let storageKey = "com.hotcar.geofences"
    
    // MARK: - Initialization
    
    override private init() {
        super.init()
        locationManager.delegate = self
        loadLocationsSync()
        requestPermissions()
    }
    
    // MARK: - Public Methods
    
    func saveLocation(_ location: SavedLocation) async {
        savedLocations.append(location)
        saveLocationsSync()
        setupGeofence(for: location)
    }
    
    func removeLocation(_ locationId: String) async {
        savedLocations.removeAll { $0.id == locationId }
        saveLocationsSync()
        removeGeofence(for: locationId)
    }
    
    // MARK: - Private Methods
    
    private func setupGeofence(for location: SavedLocation) {
        let center = CLLocationCoordinate2D(
            latitude: location.latitude,
            longitude: location.longitude
        )
        
        let region = CLCircularRegion(
            center: center,
            radius: location.radius,
            identifier: "geofence_\(location.id)"
        )
        
        region.notifyOnEntry = true
        region.notifyOnExit = true
        
        locationManager.startMonitoring(for: region)
    }
    
    private func saveLocationsSync() {
        if let encoded = try? JSONEncoder().encode(savedLocations) {
            UserDefaults.standard.set(encoded, forKey: storageKey)
        }
    }
    
    private func loadLocationsSync() {
        guard let data = UserDefaults.standard.data(forKey: storageKey),
              let decoded = try? JSONDecoder().decode([SavedLocation].self, from: data) else {
            savedLocations = []
            return
        }
        savedLocations = decoded
    }
}

// MARK: - CLLocationManagerDelegate

extension GeoFenceService: CLLocationManagerDelegate {
    func locationManager(_ manager: CLLocationManager, didEnterRegion region: CLRegion) {
        // Handle entry
    }
    
    func locationManager(_ manager: CLLocationManager, didExitRegion region: CLRegion) {
        // Handle exit
    }
}

    let id: String
    let name: String
    let latitude: Double
    let longitude: Double
    let radius: Double
    let locationType: LocationType
    let isGarage: Bool
}

    case other
```
