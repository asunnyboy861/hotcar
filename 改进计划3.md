## 🔍 问题深度分析

### 问题 1: 倒计时点击开始并没有真的倒计时 ❌

#### 现象描述
- 用户点击 TimerButton 的"Start Engine"按钮
- 按钮状态变为"Running..."
- 但时间没有倒计时，环形进度条不走动

#### 根本原因分析

**数据流追踪**：

```
用户点击 TimerButton
    ↓
HomeView.timerButton.action() 被调用
    ↓
HomeViewModel.toggleTimer() 被调用
    ↓
HomeViewModel.startTimer() 被调用
    ↓
❌ 问题：只有 print("Starting timer...")
❌ 没有实际调用 CountdownTimer.start()
```

**代码证据**：

[HomeViewModel.swift](file:///Volumes/Untitled/app/20260309/hotcar/hotcar/hotcar/Features/Home/HomeViewModel.swift#L155-L166)
```swift
// MARK: - Timer Logic

private func startTimer() {
    // TODO: Implement actual timer with countdown  ⚠️ 这里是 TODO！
    print("Starting timer for \(calculatedWarmUpTime) minutes")
}

private func stopTimer() {
    // TODO: Implement timer stop  ⚠️ 这里也是 TODO！
    print("Stopping timer")
}
```

**问题本质**：
1. HomeViewModel 没有持有 CountdownTimer 实例
2. startTimer() 和 stopTimer() 是空实现，只有 print
3. TimerButton 显示的分钟数是静态的（calculatedWarmUpTime），不会动态更新

**HomeViewModel.swift（增强）**
```swift
@MainActor
final class HomeViewModel: ObservableObject {
    
    // MARK: - Published Properties
    
    @Published var currentTemperature: Double?
    @Published var tomorrowTemperature: Double?
    @Published var locationName: String = "Loading..."
    @Published var isLoading: Bool = false
    @Published var primaryVehicle: Vehicle?
    @Published var isTimerActive: Bool = false
    @Published var fuelSaved: Double = 0.0
    @Published var fuelUsed: Double = 0.0
    
    // 新增：Timer 状态绑定
    @Published var timeRemaining: Int = 0
    @Published var timerProgress: Double = 0.0
    @Published var isTimerPaused: Bool = false
    
    // MARK: - Dependencies
    
    private let weatherService = WeatherService.shared
    private let vehicleService = VehicleService.shared
    private let widgetService = WidgetDataService.shared
    private let locationService = LocationService.shared
    private let countdownTimer = CountdownTimer()  // ✅ 新增：持有 Timer 实例
    private var cancellables = Set<AnyCancellable>()
    
    // MARK: - Initialization
    
    init() {
        setupBindings()
        loadVehicle()
        setupTimerBindings()  // ✅ 新增：设置 Timer 绑定
    }
    
    // MARK: - Setup
    
    private func setupBindings() {
        // 原有代码保持不变
        vehicleService.$vehicles
            .receive(on: RunLoop.main)
            .sink { [weak self] vehicles in
                self?.primaryVehicle = vehicles.first { $0.isPrimary } ?? vehicles.first
            }
            .store(in: &cancellables)
    }
    
    // ✅ 新增：设置 Timer 状态绑定
    private func setupTimerBindings() {
        countdownTimer.$timeRemaining
            .receive(on: RunLoop.main)
            .sink { [weak self] time in
                self?.timeRemaining = time
            }
            .store(in: &cancellables)
        
        countdownTimer.$progress
            .receive(on: RunLoop.main)
            .sink { [weak self] progress in
                self?.timerProgress = progress
            }
            .store(in: &cancellables)
        
        countdownTimer.$isPaused
            .receive(on: RunLoop.main)
            .sink { [weak self] isPaused in
                self?.isTimerPaused = isPaused
            }
            .store(in: &cancellables)
        
        countdownTimer.$isRunning
            .receive(on: RunLoop.main)
            .sink { [weak self] isRunning in
                self?.isTimerActive = isRunning
            }
            .store(in: &cancellables)
    }
    
    // MARK: - Public Methods
    
    func loadWeather() {
        // 原有代码保持不变
    }
    
    func loadVehicle() {
        Task {
            await vehicleService.loadVehicles()
        }
    }
    
    func toggleTimer() {
        if isTimerActive {
            stopTimer()
        } else {
            startTimer()
        }
    }
    
    // ✅ 修复：实现真实的 Timer 逻辑
    private func startTimer() {
        guard let vehicle = primaryVehicle,
              let temperature = currentTemperature else {
            return
        }
        
        let minutes = calculatedWarmUpTime
        
        countdownTimer.start(
            minutes: minutes,
            vehicleName: vehicle.name,
            vehicleId: vehicle.id,
            temperature: temperature
        )
        
        // 更新统计
        updateFuelStats()
    }
    
    private func stopTimer() {
        countdownTimer.stop()
    }
    
    func pauseTimer() {
        countdownTimer.pause()
    }
    
    func resumeTimer() {
        countdownTimer.resume()
    }
    
    // ✅ 新增：手动调整时间
    func adjustTimerTime(by minutes: Int) {
        let newMinutes = max(1, calculatedWarmUpTime + minutes)
        countdownTimer.reset(minutes: newMinutes)
        countdownTimer.start(
            minutes: newMinutes,
            vehicleName: primaryVehicle?.name ?? "Vehicle",
            vehicleId: primaryVehicle?.id ?? "",
            temperature: currentTemperature ?? 0
        )
    }
    
    // MARK: - Private Methods
    
    private func updateFuelStats() {
        // 从 StatisticsService 获取数据
        let stats = StatisticsService.shared.statistics
        fuelSaved = stats.totalFuelSaved
        fuelUsed = stats.totalFuelUsed
    }
    
    // ✅ 计算属性保持不变
    var calculatedWarmUpTime: Int {
        guard let temperature = currentTemperature,
              let vehicle = primaryVehicle else {
            return 5
        }
        
        return WarmUpCalculationEngine.calculateWarmUpTime(
            temperature: temperature,
            vehicleType: vehicle.type,
            engineType: vehicle.engineType,
            hasBlockHeater: vehicle.hasBlockHeater,
            isPluggedIn: vehicle.isPluggedIn
        )
    }
    
    var warmUpAdvice: String? {
        guard let temperature = currentTemperature,
              let vehicle = primaryVehicle else {
            return nil
        }
        
        return WarmUpCalculationEngine.getWarmUpAdvice(
            temperature: temperature,
            warmUpTime: calculatedWarmUpTime,
            engineType: vehicle.engineType
        )
    }
}
```

                .frame(width: 60, height: 60)

    }
